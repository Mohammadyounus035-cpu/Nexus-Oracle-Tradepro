<script type="module">
import { getOrCreateIdentity } from './identity.js';
import { canonical, canonicalHash } from './canonical.js';

// State
let identity = null;
let answers = {};
let currentQuestion = 1;
let artifact = null;

// Initialize
async function init() {
  identity = await getOrCreateIdentity();
  document.getElementById('fingerprint').textContent = identity.fingerprint;
  const statusEl = document.getElementById('identityStatus');
  statusEl.textContent = identity.isNew ? 'New Identity' : 'Persistent';
  statusEl.className = 'status ' + (identity.isNew ? 'new' : 'persistent');
}

// Navigation
function showQuestion(n) {
  document.querySelectorAll('.question-card').forEach(c => c.classList.remove('active'));
  document.querySelector(`.question-card[data-question="${n}"]`).classList.add('active');
  currentQuestion = n;
  updateProgress();
}

function updateProgress() {
  const pct = ((currentQuestion - 1) / 3) * 100;
  document.getElementById('progressFill').style.width = pct + '%';
}

// Option selection
document.querySelectorAll('.option').forEach(btn => {
  btn.addEventListener('click', () => {
    const card = btn.closest('.question-card');
    const qNum = card.dataset.question;
    card.querySelectorAll('.option').forEach(b => b.classList.remove('selected'));
    btn.classList.add('selected');
    answers[qNum] = btn.dataset.value;
    
    // Enable next button
    const nextBtn = card.querySelector('.btn-primary');
    nextBtn.disabled = false;
  });
});

// Button handlers
document.getElementById('next1').addEventListener('click', () => showQuestion(2));
document.getElementById('prev2').addEventListener('click', () => showQuestion(1));
document.getElementById('next2').addEventListener('click', () => showQuestion(3));
document.getElementById('prev3').addEventListener('click', () => showQuestion(2));

document.getElementById('submit').addEventListener('click', async () => {
  await sealArtifact();
});

// Seal artifact
async function sealArtifact() {
  const payload = {
    version: '2.0',
    timestamp: new Date().toISOString(),
    identity: identity.fingerprint,
    answers: {
      decision: answers[1],
      uncertainty: answers[2],
      change: answers[3],
    },
  };
  
  const hash = await canonicalHash(payload);
  
  // Sign with identity
  const sig = await signPayload(payload, identity.privateKey);
  
  artifact = {
    ...payload,
    hash,
    signature: sig,
    algorithm: 'Ed25519-SHA256',
  };
  
  // Display
  document.getElementById('artifactOutput').textContent = canonical(artifact);
  document.getElementById('verifyHash').textContent = hash;
  document.getElementById('verifySig').textContent = sig.slice(0, 32) + '...';
  
  document.querySelectorAll('.question-card').forEach(c => c.classList.remove('active'));
  document.getElementById('resultPanel').classList.add('active');
  document.getElementById('progressFill').style.width = '100%';
}

async function signPayload(payload, privateKey) {
  // Use Web Crypto or tweetnacl fallback
  const bytes = new TextEncoder().encode(canonical(payload));
  try {
    const key = await crypto.subtle.importKey(
      'raw', privateKey, { name: 'Ed25519' }, false, ['sign']
    );
    const sig = await crypto.subtle.sign('Ed25519', key, bytes);
    return Array.from(new Uint8Array(sig))
      .map(b => b.toString(16).padStart(2, '0'))
      .join('');
  } catch (e) {
    // Fallback to tweetnacl
    const nacl = window.nacl || await import('https://cdn.jsdelivr.net/npm/tweetnacl@1.0.3/nacl-fast.min.js');
    const sig = nacl.sign.detached(bytes, privateKey);
    return Array.from(sig).map(b => b.toString(16).padStart(2, '0')).join('');
  }
}

// UI helpers
window.toggleVerify = function() {
  document.getElementById('verifyPanel').classList.toggle('expanded');
};

window.copyArtifact = function() {
  navigator.clipboard.writeText(canonical(artifact));
  const btn = event.target;
  btn.textContent = '✓ Copied';
  setTimeout(() => btn.textContent = '📋 Copy Artifact', 2000);
};

window.downloadArtifact = function() {
  const blob = new Blob([canonical(artifact)], { type: 'application/json' });
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url;
  a.download = `dhn-artifact-${artifact.identity}.json`;
  a.click();
  URL.revokeObjectURL(url);
};

window.verifyPasted = async function() {
  const input = document.getElementById('pasteArea').value.trim();
  const resultEl = document.getElementById('pasteResult');
  
  try {
    const parsed = JSON.parse(input);
    const recomputed = await canonicalHash({
      version: parsed.version,
      timestamp: parsed.timestamp,
      identity: parsed.identity,
      answers: parsed.answers,
    });
    
    const hashValid = recomputed === parsed.hash;
    resultEl.innerHTML = `
      <div class="status-indicator ${hashValid ? 'valid' : 'invalid'}">
        <span class="icon">${hashValid ? '✓' : '✗'}</span>
        Hash ${hashValid ? 'Valid' : 'INVALID'}
      </div>
      <div style="margin-top:8px;color:#888;font-size:12px;">
        Recomputed: ${recomputed}<br>
        Stored: ${parsed.hash}
      </div>
    `;
  } catch (e) {
    resultEl.innerHTML = `<span style="color:var(--accent)">Invalid artifact format</span>`;
  }
};

init();
</script>
</body>
