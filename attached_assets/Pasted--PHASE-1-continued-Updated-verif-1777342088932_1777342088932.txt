<script type="module">
/**
 * canonical() — Same implementation as unified-test.html
 * MUST be byte-for-byte identical for hash verification to work.
 */
function canonical(obj) {
  return _canonicalize(obj);
}

function _canonicalize(obj) {
  if (obj === null) return 'null';
  if (typeof obj === 'boolean') return obj ? 'true' : 'false';
  if (typeof obj === 'number') return _canonicalNumber(obj);
  if (typeof obj === 'string') return JSON.stringify(obj);
  if (Array.isArray(obj)) {
    return '[' + obj.map(_canonicalize).join(',') + ']';
  }
  if (typeof obj === 'object') {
    const keys = Object.keys(obj).sort();
    const pairs = keys.map(k => JSON.stringify(k) + ':' + _canonicalize(obj[k]));
    return '{' + pairs.join(',') + '}';
  }
  throw new TypeError(`JCS: unsupported type ${typeof obj}`);
}

function _canonicalNumber(n) {
  if (!Number.isFinite(n)) {
    throw new Error('JCS: NaN and Infinity are not permitted');
  }
  if (Number.isInteger(n)) return String(n);
  let s = n.toFixed(20);
  s = s.replace(/0+$/, '').replace(/\.$/, '');
  if (s.includes('e') || s.includes('E')) {
    s = _expandScientific(s);
  }
  return s;
}

function _expandScientific(s) {
  const [mantissa, expStr] = s.toLowerCase().split('e');
  const exp = parseInt(expStr, 10);
  let digits = mantissa.replace('.', '');
  if (exp >= 0) {
    if (exp >= digits.length - 1) {
      return digits + '0'.repeat(exp - digits.length + 1);
    }
    return digits.slice(0, exp + 1) + '.' + digits.slice(exp + 1).replace(/0+$/, '').replace(/\.$/, '');
  } else {
    return '0.' + '0'.repeat(-exp - 1) + digits;
  }
}

async function canonicalHash(obj) {
  const bytes = new TextEncoder().encode(canonical(obj));
  const hashBuffer = await crypto.subtle.digest('SHA-256', bytes);
  return Array.from(new Uint8Array(hashBuffer))
    .map(b => b.toString(16).padStart(2, '0'))
    .join('');
}

/**
 * Verify Ed25519 signature using tweetnacl
 */
async function verifySignature(messageBytes, signatureHex, publicKeyHex) {
  const nacl = window.nacl || await loadTweetNacl();
  const signature = hexToBytes(signatureHex);
  const publicKey = hexToBytes(publicKeyHex);
  return nacl.sign.detached.verify(messageBytes, signature, publicKey);
}

async function loadTweetNacl() {
  if (typeof window !== 'undefined' && window.nacl) return window.nacl;
  // Load from CDN
  return new Promise((resolve, reject) => {
    const script = document.createElement('script');
    script.src = 'https://cdn.jsdelivr.net/npm/tweetnacl@1.0.3/nacl-fast.min.js';
    script.onload = () => resolve(window.nacl);
    script.onerror = reject;
    document.head.appendChild(script);
  });
}

function hexToBytes(hex) {
  const bytes = new Uint8Array(hex.length / 2);
  for (let i = 0; i < hex.length; i += 2) {
    bytes[i / 2] = parseInt(hex.slice(i, i + 2), 16);
  }
  return bytes;
}

// Main verification
window.verifyArtifact = async function() {
  const input = document.getElementById('artifactInput').value.trim();
  const errorEl = document.getElementById('errorMsg');
  const resultsEl = document.getElementById('results');
  
  errorEl.classList.remove('active');
  resultsEl.classList.remove('active');
  
  try {
    const artifact = JSON.parse(input);
    
    // Validate structure
    if (!artifact.version || !artifact.timestamp || !artifact.identity || !artifact.answers) {
      throw new Error('Missing required fields in artifact');
    }
    
    // Recompute hash
    const payload = {
      version: artifact.version,
      timestamp: artifact.timestamp,
      identity: artifact.identity,
      answers: artifact.answers,
    };
    const recomputedHash = await canonicalHash(payload);
    const hashValid = recomputedHash === artifact.hash;
    
    // Verify signature (requires public key — in practice, this would be resolved
    // from identity fingerprint via a registry or the artifact would include pubkey)
    let sigValid = false;
    let sigChecked = false;
    
    if (artifact.signature && artifact.publicKey) {
      const messageBytes = new TextEncoder().encode(canonical(payload));
      sigValid = await verifySignature(messageBytes, artifact.signature, artifact.publicKey);
      sigChecked = true;
    }
    
    // Update UI
    updateStatusCard('hash', hashValid);
    updateStatusCard('sig', sigValid, !sigChecked);
    
    document.getElementById('detailVersion').textContent = artifact.version;
    document.getElementById('detailTimestamp').textContent = artifact.timestamp;
    document.getElementById('detailIdentity').textContent = artifact.identity;
    document.getElementById('detailAnswers').textContent = 
      `Decision: ${artifact.answers.decision}, Uncertainty: ${artifact.answers.uncertainty}, Change: ${artifact.answers.change}`;
    document.getElementById('detailAlgo').textContent = artifact.algorithm || 'Unknown';
    document.getElementById('detailHash').textContent = artifact.hash;
    document.getElementById('detailSig').textContent = artifact.signature ? 
      (artifact.signature.slice(0, 48) + '...') : 'Not provided';
    
    resultsEl.classList.add('active');
    
  } catch (e) {
    errorEl.textContent = 'Error: ' + e.message;
    errorEl.classList.add('active');
  }
};

function updateStatusCard(type, valid, unchecked = false) {
  const card = document.getElementById(type + 'Card');
  const icon = document.getElementById(type + 'Icon');
  const value = document.getElementById(type + 'Value');
  
  card.className = 'status-card ' + (valid ? 'valid' : 'invalid');
  
  if (unchecked) {
    icon.textContent = '?';
    value.textContent = 'No Key';
    card.className = 'status-card';
    return;
  }
  
  if (valid) {
    icon.textContent = '✓';
    value.textContent = 'Valid';
  } else {
    icon.textContent = '✗';
    value.textContent = 'Invalid';
  }
}
