export interface HistoricalSnapshot {
  symbol: string;
  returns: number;        // %
  volatility: number;     // %
  volume: number;
  timestamp: string;
}

export interface LiveSnapshot extends HistoricalSnapshot {}

export interface LatticeState {
  pathId: number;         // 1–96
  polarity: -1 | 0 | 1;
  strength: number;       // 0–1
  guardian: 'lion' | 'phoenix' | 'dragon' | 'owl' | 'raven';
}

export interface AnalysisOutput {
  symbol: string;
  historicalState: LatticeState;
  liveState: LatticeState;
  drift: number;              // deviation magnitude
  regime: 'stable' | 'transition' | 'volatile' | 'anomaly';
  hash: string;
}

// ---------------------------------------------------------------------------
// FEATURE EXTRACTION
// ---------------------------------------------------------------------------

export function computeFeatures(prices: number[]): {
  returns: number;
  volatility: number;
} {
  if (prices.length < 2) return { returns: 0, volatility: 0 };

  const returns = (prices[prices.length - 1] - prices[0]) / prices[0];

  const diffs = [];
  for (let i = 1; i < prices.length; i++) {
    diffs.push((prices[i] - prices[i - 1]) / prices[i - 1]);
  }

  const mean = diffs.reduce((a, b) => a + b, 0) / diffs.length;
  const variance =
    diffs.reduce((sum, d) => sum + Math.pow(d - mean, 2), 0) /
    diffs.length;

  return {
    returns,
    volatility: Math.sqrt(variance),
  };
}

// ---------------------------------------------------------------------------
// 96 PATH MAPPING
// ---------------------------------------------------------------------------

export function mapToLattice(features: {
  returns: number;
  volatility: number;
  volume: number;
}): LatticeState {
  // Normalize inputs
  const r = Math.tanh(features.returns * 5);
  const v = Math.tanh(features.volatility * 10);
  const vol = Math.tanh(features.volume / 10000);

  // Combine into scalar field
  const score = (r * 0.5 + v * 0.3 + vol * 0.2);

  // Map to 1–96
  const pathId = Math.min(
    96,
    Math.max(1, Math.floor(((score + 1) / 2) * 96))
  );

  // Polarity
  const polarity = score > 0.2 ? 1 : score < -0.2 ? -1 : 0;

  // Guardian mapping (based on segments)
  const guardian =
    pathId <= 19
      ? 'lion'
      : pathId <= 38
      ? 'phoenix'
      : pathId <= 57
      ? 'dragon'
      : pathId <= 76
      ? 'owl'
      : 'raven';

  return {
    pathId,
    polarity,
    strength: Math.abs(score),
    guardian,
  };
}

// ---------------------------------------------------------------------------
// DRIFT DETECTION
// ---------------------------------------------------------------------------

export function computeDrift(
  hist: LatticeState,
  live: LatticeState
): number {
  const pathDiff = Math.abs(hist.pathId - live.pathId) / 96;
  const polarityDiff = hist.polarity === live.polarity ? 0 : 1;
  const strengthDiff = Math.abs(hist.strength - live.strength);

  return pathDiff * 0.5 + polarityDiff * 0.3 + strengthDiff * 0.2;
}

// ---------------------------------------------------------------------------
// REGIME CLASSIFICATION
// ---------------------------------------------------------------------------

export function classifyRegime(drift: number): AnalysisOutput['regime'] {
  if (drift < 0.15) return 'stable';
  if (drift < 0.35) return 'transition';
  if (drift < 0.65) return 'volatile';
  return 'anomaly';
}

// ---------------------------------------------------------------------------
// FULL ANALYSIS PIPELINE
// ---------------------------------------------------------------------------

export async function runAnalysis(params: {
  symbol: string;
  historicalPrices: number[];
  livePrices: number[];
  volume: number;
}): Promise<AnalysisOutput> {
  const histFeatures = computeFeatures(params.historicalPrices);
  const liveFeatures = computeFeatures(params.livePrices);

  const historicalState = mapToLattice({
    ...histFeatures,
    volume: params.volume,
  });

  const liveState = mapToLattice({
    ...liveFeatures,
    volume: params.volume,
  });

  const drift = computeDrift(historicalState, liveState);
  const regime = classifyRegime(drift);

  const payload = {
    symbol: params.symbol,
    historicalState,
    liveState,
    drift,
    regime,
  };

  const hash = await canonicalHash(payload);

  return {
    ...payload,
    hash,
  };
}

// ---------------------------------------------------------------------------
// SIGNAL LOGIC
// ---------------------------------------------------------------------------

export function generateSignal(analysis: AnalysisOutput): TradeSignal {
  const { liveState, drift, regime } = analysis;

  let action: TradeAction = 'hold';
  let confidence = 0;
  let size = 0;
  let reason = '';

  // -----------------------------------------------------------------------
  // CORE DECISION MATRIX
  // -----------------------------------------------------------------------

  if (regime === 'stable') {
    if (liveState.polarity === 1) {
      action = 'buy';
      confidence = 0.7 + liveState.strength * 0.3;
      size = 0.5 + liveState.strength * 0.5;
      reason = 'Stable + positive polarity';
    }

    if (liveState.polarity === -1) {
      action = 'sell';
      confidence = 0.7 + liveState.strength * 0.3;
      size = 0.5 + liveState.strength * 0.5;
      reason = 'Stable + negative polarity';
    }
  }

  if (regime === 'transition') {
    action = 'hold';
    confidence = 0.4;
    size = 0.2;
    reason = 'Transition phase — reduced exposure';
  }

  if (regime === 'volatile') {
    action = 'hold';
    confidence = 0.2;
    size = 0.1;
    reason = 'High volatility — capital preservation';
  }

  if (regime === 'anomaly') {
    action = 'hold';
    confidence = 0;
    size = 0;
    reason = 'Anomaly detected — no trade';
  }

  return {
    symbol: analysis.symbol,
    action,
    confidence,
    size,
    reason,
    regime,
    pathId: liveState.pathId,
  };
}

import { TradeSignal } from './signalEngine';

export interface ValidationResult {
  approved: boolean;
  reason?: string;
}

export function validateSignal(signal: TradeSignal): ValidationResult {
  // Γ1 — Owl (logic consistency)
  if (signal.confidence < 0 || signal.confidence > 1) {
    return { approved: false, reason: 'Invalid confidence bounds' };
  }

  // Γ2 — Raven (anomaly detection)
  if (signal.regime === 'anomaly') {
    return { approved: false, reason: 'Blocked: anomaly regime' };
  }

  // Γ3 — Phoenix (adaptation check)
  if (signal.regime === 'transition' && signal.size > 0.3) {
    return { approved: false, reason: 'Position too large in transition' };
  }

  // Γ4 — Dragon (energy/momentum)
  if (signal.confidence < 0.3 && signal.action !== 'hold') {
    return { approved: false, reason: 'Insufficient signal strength' };
  }

  // Γ5 — Lion (final authority)
  return { approved: true };
}

/**
 * executionBridge.ts
 * 
 * Connects signal engine → trade engine
 */

import { TradeEngine } from './tradeEngine';
import { generateSignal } from './signalEngine';
import { validateSignal } from './sovereignGuardians';
import { AnalysisOutput } from './marketAnalysisFlow';

const engine = new TradeEngine();

export function processAnalysis(analysis: AnalysisOutput) {
  const signal = generateSignal(analysis);

  const validation = validateSignal(signal);

  if (!validation.approved) {
    return {
      status: 'rejected',
      reason: validation.reason,
      signal,
    };
  }

  if (signal.action === 'hold') {
    return {
      status: 'no_action',
      signal,
    };
  }

  const order = engine.submitOrder({
    symbol: signal.symbol,
    side: signal.action,
    orderType: 'market',
    shares: Math.floor(signal.size * 100), // scaled sizing
  });

  return {
    status: 'executed',
    signal,
    order,
  };
}

// ---------------------------------------------------------------------------
// Layer 1 — Drift penalty
// ---------------------------------------------------------------------------

export function applyDriftPenalty(
  signal: TradeSignal,
  analysis: AnalysisOutput
): TradeSignal {
  const penalty = Math.max(0, 1 - analysis.drift);

  return {
    ...signal,
    confidence: signal.confidence * penalty,
    size: signal.size * penalty,
    reason: signal.reason + '|drift_adjusted',
  };
}

// ---------------------------------------------------------------------------
// Layer 2 — Path stability weighting
// ---------------------------------------------------------------------------

export function applyPathStability(
  signal: TradeSignal,
  analysis: AnalysisOutput
): TradeSignal {
  const delta =
    Math.abs(
      analysis.liveState.pathId - analysis.historicalState.pathId
    ) / 96;

  const stability = 1 - delta;

  return {
    ...signal,
    confidence: signal.confidence * stability,
    size: signal.size * stability,
    reason: signal.reason + '|path_stability',
  };
}

// ---------------------------------------------------------------------------
// Layer 3 — Strength normalization clamp
// ---------------------------------------------------------------------------

export function normalizeSignal(signal: TradeSignal): TradeSignal {
  return {
    ...signal,
    confidence: Math.min(1, Math.max(0, signal.confidence)),
    size: Math.min(1, Math.max(0, signal.size)),
    reason: signal.reason + '|normalized',
  };
}

import { TradeSignal, ValidatedSignal } from './layerTypes';

export function validateSignalStrict(signal: TradeSignal): ValidatedSignal {
  // Γ1 — bounds
  if (signal.confidence < 0 || signal.confidence > 1) {
    return { ...signal, approved: false, rejectionReason: 'invalid_confidence' };
  }

  // Γ2 — anomaly block
  if (signal.regime === 'anomaly') {
    return { ...signal, approved: false, rejectionReason: 'anomaly_block' };
  }

  // Γ3 — weak signal suppression
  if (signal.confidence < 0.25 && signal.action !== 'hold') {
    return { ...signal, approved: false, rejectionReason: 'low_confidence' };
  }

  // Γ4 — transition sizing restriction
  if (signal.regime === 'transition' && signal.size > 0.25) {
    return { ...signal, approved: false, rejectionReason: 'oversize_transition' };
  }

  // Γ5 — volatile enforcement
  if (signal.regime === 'volatile' && signal.action !== 'hold') {
    return { ...signal, approved: false, rejectionReason: 'volatile_block' };
  }

  return { ...signal, approved: true };
}

import { RiskEnvelope, TradeSignal } from './layerTypes';

export function getRiskEnvelope(regime: string): RiskEnvelope {
  if (regime === 'stable') {
    return { maxPosition: 1, maxDrawdown: 0.25, volatilityCap: 1 };
  }
  if (regime === 'transition') {
    return { maxPosition: 0.3, maxDrawdown: 0.15, volatilityCap: 0.6 };
  }
  if (regime === 'volatile') {
    return { maxPosition: 0.1, maxDrawdown: 0.1, volatilityCap: 0.3 };
  }
  return { maxPosition: 0, maxDrawdown: 0.05, volatilityCap: 0.1 };
}

export function applyRiskEnvelope(
  signal: TradeSignal,
  envelope: RiskEnvelope
): TradeSignal {
  return {
    ...signal,
    size: Math.min(signal.size, envelope.maxPosition),
    reason: signal.reason + '|risk_enforced',
  };
}

import { ExecutionDecision, ValidatedSignal } from './layerTypes';

export function decideExecution(
  signal: ValidatedSignal
): ExecutionDecision {
  if (!signal.approved) {
    return {
      execute: false,
      shares: 0,
      reason: signal.rejectionReason || 'rejected',
    };
  }

  if (signal.action === 'hold') {
    return {
      execute: false,
      shares: 0,
      reason: 'hold',
    };
  }

  return {
    execute: true,
    shares: Math.floor(signal.size * 100),
    side: signal.action,
    reason: 'execute',
  };
}

import { TradeSignal } from './layerTypes';

export interface MicrostructureInput {
  bidDepth: number;
  askDepth: number;
  spread: number;       // cents
  imbalance: number;    // -1 to +1
}

export function applyMicrostructure(
  signal: TradeSignal,
  micro: MicrostructureInput
): TradeSignal {
  const pressure = micro.imbalance;

  let bias = 1;

  if (signal.action === 'buy') {
    bias = 1 + pressure;
  } else if (signal.action === 'sell') {
    bias = 1 - pressure;
  }

  const spreadPenalty = Math.max(0, 1 - micro.spread / 100);

  return {
    ...signal,
    confidence: signal.confidence * bias * spreadPenalty,
    size: signal.size * bias * spreadPenalty,
    reason: signal.reason + '|microstructure',
  };
}

export function applyTemporalConsistency(
  signal: TradeSignal,
  previousSignals: TradeSignal[]
): TradeSignal {
  if (previousSignals.length === 0) return signal;

  const last = previousSignals[previousSignals.length - 1];

  const consistency =
    signal.action === last.action ? 1.1 : 0.7;

  return {
    ...signal,
    confidence: Math.min(1, signal.confidence * consistency),
    size: signal.size * consistency,
    reason: signal.reason + '|temporal',
  };
}

// ============================================================================
// stableStringify.ts — deterministic stringify (sorted keys)
// ============================================================================

export function stableStringify(value: any): string {
  return _stringify(value);

  function _stringify(v: any): string {
    if (v === null) return 'null';
    if (typeof v === 'number') {
      if (!Number.isFinite(v)) throw new Error('non-finite number');
      return String(v);
    }
    if (typeof v === 'boolean') return v ? 'true' : 'false';
    if (typeof v === 'string') return JSON.stringify(v);
    if (Array.isArray(v)) {
      return '[' + v.map(_stringify).join(',') + ']';
    }
    if (typeof v === 'object') {
      const keys = Object.keys(v).sort();
      const parts: string[] = [];
      for (const k of keys) {
        parts.push(JSON.stringify(k) + ':' + _stringify(v[k]));
      }
      return '{' + parts.join(',') + '}';
    }
    throw new Error('unsupported type');
  }
}


// ============================================================================
// deterministicHash.ts — SHA-256 over stable stringify
// ============================================================================

export async function sha256Hex(input: string): Promise<string> {
  const data = new TextEncoder().encode(input);
  const buf = await crypto.subtle.digest('SHA-256', data);
  const arr = Array.from(new Uint8Array(buf));
  return arr.map(b => b.toString(16).padStart(2, '0')).join('');
}

export async function hashObject(obj: any): Promise<string> {
  return sha256Hex(stableStringify(obj));
}

import { TradeSignal, ValidatedSignal, ExecutionDecision } from './layerTypes';

export function assertSignalInvariants(s: TradeSignal): void {
  if (s.confidence < 0 || s.confidence > 1) {
    throw new Error('INV:confidence_out_of_bounds');
  }
  if (s.size < 0 || s.size > 1) {
    throw new Error('INV:size_out_of_bounds');
  }
  if (!['buy', 'sell', 'hold'].includes(s.action)) {
    throw new Error('INV:invalid_action');
  }
}

export function assertValidatedInvariants(v: ValidatedSignal): void {
  if (v.approved && v.rejectionReason) {
    throw new Error('INV:approved_with_reason');
  }
  if (!v.approved && !v.rejectionReason) {
    throw new Error('INV:rejected_without_reason');
  }
}

export function assertDecisionInvariants(d: ExecutionDecision): void {
  if (!d.execute && d.shares !== 0) {
    throw new Error('INV:nonzero_shares_when_not_execute');
  }
  if (d.execute && (!d.side || d.shares <= 0)) {
    throw new Error('INV:execute_requires_side_and_positive_shares');
  }
}


// ============================================================================
// freeze.ts — immutability guard
// ============================================================================

export function deepFreeze<T>(obj: T): T {
  if (obj && typeof obj === 'object') {
    Object.freeze(obj);
    for (const k of Object.keys(obj as any)) {
      // @ts-ignore
      const v = obj[k];
      if (v && typeof v === 'object' && !Object.isFrozen(v)) {
        deepFreeze(v);
      }
    }
  }
  return obj;
}

export async function cacheGetOrCompute<T>(

  compute: () => Promise<T> | T
): Promise<T> {
  const key = await hashObject(keyObj);
  if (_cache.has(key)) return _cache.get(key);
  const val = await compute();
  _cache.set(key, val);
  return val;
}

// ============================================================================
// signature.ts — Ed25519 sign/verify using Web Crypto
// ============================================================================

export async function signPayload(
  privateKey: CryptoKey,
  payload: any
): Promise<string> {
  const data = new TextEncoder().encode(stableStringify(payload));
  const sig = await crypto.subtle.sign('Ed25519', privateKey, data);
  const arr = Array.from(new Uint8Array(sig));
  return arr.map(b => b.toString(16).padStart(2, '0')).join('');
}

export async function verifyPayload(
  publicKey: CryptoKey,
  payload: any,
  signatureHex: string
): Promise<boolean> {
  const data = new TextEncoder().encode(stableStringify(payload));
  const sigBytes = new Uint8Array(
    signatureHex.match(/.{1,2}/g)!.map(h => parseInt(h, 16))
  );
  return crypto.subtle.verify('Ed25519', publicKey, sigBytes, data);
}

import { AnalysisOutput } from './layerTypes';

  assertSignalInvariants,
  assertValidatedInvariants,
  assertDecisionInvariants,
} from './invariants';

import { cacheGetOrCompute } from './cache';
import { reportMetric } from './metricsHook';
import { hashObject } from './deterministicHash';

export async function runHardenedPipeline(
  analysis: AnalysisOutput,
  context: any
) {
  return cacheGetOrCompute({ analysis, context }, async () => {
    try {
      const out = runPhase2Extended(analysis, context);

      assertSignalInvariants(out.signal);
      assertValidatedInvariants(out.decision.execute
        ? { ...out.signal, approved: true }
        : { ...out.signal, approved: false, rejectionReason: out.decision.reason }
      );
      assertDecisionInvariants(out.decision);

      const hash = await hashObject(out);

      const frozen = deepFreeze({
        ...out,
        hash,
      });

      return frozen;
    } catch (e: any) {
      reportMetric('pipeline_error', { error: String(e) });
      throw e;
    }
  });
}

// ============================================================================
// stateMachine.ts — deterministic execution state machine
// ============================================================================

export type EngineState =
  | 'IDLE'
  | 'ANALYZE'
  | 'SIGNAL'
  | 'VALIDATE'
  | 'DECIDE'
  | 'EXECUTE'
  | 'REJECT'
  | 'HALT';

export interface StateContext {
  state: EngineState;
  lastTransition: string;
}

export function transitionState(
  ctx: StateContext,
  event: string
): StateContext {
  const { state } = ctx;

  if (state === 'IDLE' && event === 'start') {
    return { state: 'ANALYZE', lastTransition: 'IDLE→ANALYZE' };
  }

  if (state === 'ANALYZE' && event === 'analysis_ready') {
    return { state: 'SIGNAL', lastTransition: 'ANALYZE→SIGNAL' };
  }

  if (state === 'SIGNAL' && event === 'signal_ready') {
    return { state: 'VALIDATE', lastTransition: 'SIGNAL→VALIDATE' };
  }

  if (state === 'VALIDATE' && event === 'approved') {
    return { state: 'DECIDE', lastTransition: 'VALIDATE→DECIDE' };
  }

  if (state === 'VALIDATE' && event === 'rejected') {
    return { state: 'REJECT', lastTransition: 'VALIDATE→REJECT' };
  }

  if (state === 'DECIDE' && event === 'execute') {
    return { state: 'EXECUTE', lastTransition: 'DECIDE→EXECUTE' };
  }

  if (state === 'DECIDE' && event === 'skip') {
    return { state: 'IDLE', lastTransition: 'DECIDE→IDLE' };
  }

  if (state === 'EXECUTE' && event === 'done') {
    return { state: 'IDLE', lastTransition: 'EXECUTE→IDLE' };
  }

  return { state: 'HALT', lastTransition: 'INVALID_TRANSITION' };
}

import { Vector } from './vectorTypes';

export function isValidVector(v: Vector): boolean {
  // Law 1: no high risk in dev
  if (v.env === 'dev' && v.risk === 'high') return false;

  // Law 2: rejected cannot execute
  if (v.auth === 'rejected' && v.decision !== 'hold') return false;

  // Law 3: live requires approval
  if (v.mode === 'live' && v.auth !== 'approved') return false;

  return true;
}

import { Vector, Env, Mode, Risk, Decision, Auth } from './vectorTypes';
import { isValidVector } from './constraints';

export function generateAllVectors(): Vector[] {
  const envs: Env[] = ['prod', 'staging', 'dev'];
  const modes: Mode[] = ['live', 'paper', 'backtest'];
  const risks: Risk[] = ['low', 'medium', 'high'];
  const decisions: Decision[] = ['buy', 'sell', 'hold'];
  const auths: Auth[] = ['approved', 'rejected'];

  const result: Vector[] = [];

  for (const env of envs)
    for (const mode of modes)
      for (const risk of risks)
        for (const decision of decisions)
          for (const auth of auths) {
            const v: Vector = { env, mode, risk, decision, auth };
            if (isValidVector(v)) result.push(v);
          }

  return result;
}

// ============================================================================
// consensusLayer.ts — multi-signal agreement
// ============================================================================

import { TradeSignal } from './layerTypes';

export function consensusDecision(signals: TradeSignal[]): TradeSignal {
  if (signals.length === 0) throw new Error('no_signals');

  let score = 0;

  for (const s of signals) {
    if (s.action === 'buy') score += s.confidence;
    if (s.action === 'sell') score -= s.confidence;
  }

  const avg = score / signals.length;

  const base = signals[0];

  return {
    ...base,
    action: avg > 0.2 ? 'buy' : avg < -0.2 ? 'sell' : 'hold',
    confidence: Math.abs(avg),
    size: Math.min(1, Math.abs(avg)),
    reason: 'consensus',
  };
}


// ============================================================================
// multiPathFusion.ts — 96-path aggregation
// ============================================================================

import { LatticeState } from './layerTypes';

export function fuseLatticeStates(states: LatticeState[]): LatticeState {
  if (states.length === 0) throw new Error('no_states');

  let total = 0;
  let strength = 0;

  for (const s of states) {
    total += s.polarity * s.strength;
    strength += s.strength;
  }

  const avg = total / states.length;

  return {
    pathId: Math.round(
      states.reduce((sum, s) => sum + s.pathId, 0) / states.length
    ),
    polarity: avg > 0.1 ? 1 : avg < -0.1 ? -1 : 0,
    strength: Math.min(1, strength / states.length),
    guardian: states[0].guardian,
  };
}

  if (signal.confidence > 0.6) score += 0.3;      // Lion
  if (signal.size < 0.5) score += 0.2;            // Phoenix
  if (signal.confidence > 0.5) score += 0.2;      // Dragon
  if (signal.action === 'hold') score += 0.1;     // Owl

export async function supervisorLoop(
  analyses: AnalysisOutput[],
  contextFactory: (i: number) => any
) {
  const outputs: any[] = [];

  for (let i = 0; i < analyses.length; i++) {
    const ctx = contextFactory(i);

    const halt = checkFailSafe({
      drawdown: ctx.drawdown,
      volatility: ctx.volatility,
      anomaly: analyses[i].regime === 'anomaly',

    }

    const out = await runHardenedPipeline(analyses[i], ctx);
    outputs.push(out);
  }

  return outputs;
}

// portfolio.ts — minimal deterministic portfolio
// ============================================================================

export interface Position {
  symbol: string;
  shares: number;
  avgPrice: number;
}

export class Portfolio {
  cash: number;
  positions: Map<string, Position>;
  history: number[];

  constructor(initialCash: number) {
    this.cash = initialCash;
    this.positions = new Map();
    this.history = [initialCash];
  }

  updateEquity(priceMap: Record<string, number>) {
    let total = this.cash;

    for (const pos of this.positions.values()) {
      total += pos.shares * priceMap[pos.symbol];
    }

    this.history.push(total);
  }
}


// ============================================================================
// executionSimulator.ts
// ============================================================================

import { Portfolio } from './portfolio';

export function simulateExecution(
  portfolio: Portfolio,
  decision: any,
  price: number
) {
  if (!decision.execute) return;

  const shares = decision.shares;

  if (decision.side === 'buy') {
    const cost = shares * price;
    if (portfolio.cash >= cost) {
      portfolio.cash -= cost;

      const pos = portfolio.positions.get('SYM') || {
        symbol: 'SYM',
        shares: 0,
        avgPrice: price,
      };

      const totalShares = pos.shares + shares;

      pos.avgPrice =
        (pos.avgPrice * pos.shares + price * shares) / totalShares;

      pos.shares = totalShares;

      portfolio.positions.set('SYM', pos);
    }
  }

  if (decision.side === 'sell') {
    const pos = portfolio.positions.get('SYM');
    if (!pos || pos.shares < shares) return;

    portfolio.cash += shares * price;
    pos.shares -= shares;

    if (pos.shares === 0) {
      portfolio.positions.delete('SYM');
    }
  }
}


// ============================================================================
// backtestEngine.ts
// ============================================================================

import { AnalysisOutput } from './layerTypes';
import { finalizePhase2 } from './finalPhase2';
import { Portfolio } from './portfolio';
import { simulateExecution } from './executionSimulator';

export async function runBacktest(
  analyses: AnalysisOutput[],
  contextFactory: (i: number) => any,
  priceSeries: number[],
  privateKey: CryptoKey
): Promise<BacktestResult> {
  const portfolio = new Portfolio(100000);
  let wins = 0;
  let losses = 0;

  for (let i = 0; i < analyses.length; i++) {
    const ctx = contextFactory(i);

    const artifact = await finalizePhase2(
      analyses[i],
      ctx,
      privateKey
    );

    const price = priceSeries[i] || priceSeries[priceSeries.length - 1];

    simulateExecution(portfolio, artifact.decision, price);

    portfolio.updateEquity({ SYM: price });

    if (portfolio.history.length > 1) {
      const diff =
        portfolio.history[portfolio.history.length - 1] -
        portfolio.history[portfolio.history.length - 2];

      if (diff > 0) wins++;
      if (diff < 0) losses++;
    }
  }

  const pnl =
    portfolio.history[portfolio.history.length - 1] - portfolio.history[0];

  let peak = portfolio.history[0];
  let maxDD = 0;

  for (const v of portfolio.history) {
    if (v > peak) peak = v;
    const dd = peak - v;
    if (dd > maxDD) maxDD = dd;
  }

  return {
    trades: analyses.length,
    wins,
    losses,
    pnl,
    equityCurve: portfolio.history,
    drawdown: maxDD,
  };
}

  let price = 10000;

  for (let i = 0; i < length; i++) {

    price = Math.max(1, Math.floor(price * (1 + rand * 0.02)));
    series.push(price);
  }

  return series;
}

// ============================================================================
// regimeBuckets.ts — group performance by regime
// ============================================================================

import { AnalysisOutput } from './layerTypes';

export interface RegimeBucket {
  regime: string;
  pnl: number;
  trades: number;
}

export function computeRegimeBuckets(
  analyses: AnalysisOutput[],
  equityCurve: number[]
): RegimeBucket[] {
  const buckets: Record<string, RegimeBucket> = {};

  for (let i = 1; i < analyses.length; i++) {
    const regime = analyses[i].regime;
    const diff = equityCurve[i] - equityCurve[i - 1];

    if (!buckets[regime]) {
      buckets[regime] = { regime, pnl: 0, trades: 0 };
    }

    buckets[regime].pnl += diff;
    buckets[regime].trades += 1;
  }

  return Object.values(buckets);
}


// ============================================================================
// advancedMetrics.ts — deeper analytics
// ============================================================================

export interface AdvancedMetrics {
  sharpe: number;
  sortino: number;
  winRate: number;
  expectancy: number;
}

export function computeAdvancedMetrics(
  equity: number[]
): AdvancedMetrics {
  const returns: number[] = [];

  for (let i = 1; i < equity.length; i++) {
    returns.push((equity[i] - equity[i - 1]) / equity[i - 1]);
  }

  const mean =
    returns.reduce((a, b) => a + b, 0) / returns.length;

  const variance =
    returns.reduce((s, r) => s + (r - mean) ** 2, 0) /
    returns.length;

  const downside =
    returns
      .filter(r => r < 0)
      .reduce((s, r) => s + r ** 2, 0) /
    returns.length;

  const sharpe = mean / Math.sqrt(variance || 1e-9);
  const sortino = mean / Math.sqrt(downside || 1e-9);

  const wins = returns.filter(r => r > 0).length;
  const winRate = wins / returns.length;

  const expectancy = mean;

  return {
    sharpe,
    sortino,
    winRate,
    expectancy,
  };
}


// ============================================================================
// monteCarlo.ts — randomized scenario testing
// ============================================================================

export function monteCarloSim(
  baseSeries: number[],
  runs: number
): number[][] {
  const simulations: number[][] = [];

  for (let r = 0; r < runs; r++) {
    const sim: number[] = [];
    let price = baseSeries[0];

    for (let i = 0; i < baseSeries.length; i++) {
      const noise = (Math.random() - 0.5) * 0.03;
      price = Math.max(1, price * (1 + noise));
      sim.push(Math.floor(price));
    }

    simulations.push(sim);
  }

  return simulations;
}

export interface Score {
  sharpe: number;
  drawdown: number;
  pnl: number;
  score: number;
}

export function computeScore(
  sharpe: number,
  drawdown: number,
  pnl: number
): Score {
  const score = sharpe * 0.6 + pnl * 0.0001 - drawdown * 0.4;
  return { sharpe, drawdown, pnl, score };
}


// ============================================================================
// kfold.ts — time-series k-fold split (no leakage)
// ============================================================================

export function kfoldIndices(
  n: number,
  k: number
): { train: number[]; test: number[] }[] {
  const foldSize = Math.floor(n / k);
  const folds: { train: number[]; test: number[] }[] = [];

  for (let i = 0; i < k; i++) {
    const start = i * foldSize;
    const end = i === k - 1 ? n : start + foldSize;

    const test: number[] = [];
    const train: number[] = [];

    for (let j = 0; j < n; j++) {
      if (j >= start && j < end) test.push(j);
      else train.push(j);
    }

    folds.push({ train, test });
  }

  return folds;
}


// ============================================================================
// optimizer.ts — grid/random search over params
// ============================================================================

import { AnalysisOutput } from './layerTypes';
import { runPhase3 } from './phase3Orchestrator';
import { Params } from './params';
import { computeScore, Score } from './score';

export interface OptimizationResult {
  params: Params;
  score: Score;
}

export async function optimizeParams(
  analyses: AnalysisOutput[],
  contextFactory: (i: number, p: Params) => any,
  privateKey: CryptoKey,
  paramSet: Params[]
): Promise<OptimizationResult[]> {
  const results: OptimizationResult[] = [];

  for (const p of paramSet) {
    const res = await runPhase3(
      analyses,
      (i) => contextFactory(i, p),
      privateKey
    );

    const sharpe = 0; // placeholder if not provided
    const pnl = res.pnl;
    const drawdown = res.drawdown;

    const score = computeScore(sharpe, drawdown, pnl);

    results.push({ params: p, score });
  }

  return results.sort((a, b) => b.score.score - a.score.score);
}


// ============================================================================
// crossValidation.ts — k-fold evaluation
// ============================================================================

import { kfoldIndices } from './kfold';
import { optimizeParams } from './optimizer';
import { Params } from './params';
import { AnalysisOutput } from './layerTypes';

export async function crossValidate(
  analyses: AnalysisOutput[],
  contextFactory: (i: number, p: Params) => any,
  privateKey: CryptoKey,
  paramSet: Params[],
  k: number
) {
  const folds = kfoldIndices(analyses.length, k);
  const results: any[] = [];

  for (const fold of folds) {
    const train = fold.train.map(i => analyses[i]);
    const test = fold.test.map(i => analyses[i]);

    const opt = await optimizeParams(
      train,
      contextFactory,
      privateKey,
      paramSet
    );

    const best = opt[0];

    const testRes = await optimizeParams(
      test,
      contextFactory,
      privateKey,
      [best.params]
    );

    results.push({
      trainBest: best,
      testScore: testRes[0],
    });
  }

  return results;
}


// ============================================================================
// robustness.ts — stability across perturbations
// ============================================================================

export function robustnessScore(scores: number[]): number {
  const mean = scores.reduce((a, b) => a + b, 0) / scores.length;
  const variance =
    scores.reduce((s, x) => s + (x - mean) ** 2, 0) / scores.length;

  return mean / (1 + Math.sqrt(variance));
}

// ============================================================================
// ensemble.ts — combine multiple parameterized runs
// ============================================================================

import { Params } from './params';
import { AnalysisOutput } from './layerTypes';
import { runPhase3 } from './phase3Orchestrator';

export interface EnsembleMember {
  params: Params;
  weight: number; // 0–1
}

export interface EnsembleResult {
  equityCurve: number[];
  pnl: number;
  drawdown: number;
}

export async function runEnsemble(
  analyses: AnalysisOutput[],
  contextFactory: (i: number, p: Params) => any,
  privateKey: CryptoKey,
  members: EnsembleMember[]
): Promise<EnsembleResult> {
  const curves: number[][] = [];

  for (const m of members) {
    const res = await runPhase3(
      analyses,
      (i) => contextFactory(i, m.params),
      privateKey
    );
    curves.push(res.equityCurve.map(v => v * m.weight));
  }

  const length = curves[0].length;
  const combined: number[] = [];

  for (let i = 0; i < length; i++) {
    let sum = 0;
    for (const c of curves) sum += c[i];
    combined.push(sum);
  }

  const pnl = combined[combined.length - 1] - combined[0];

  let peak = combined[0];
  let maxDD = 0;
  for (const v of combined) {
    if (v > peak) peak = v;
    const dd = peak - v;
    if (dd > maxDD) maxDD = dd;
  }

  return {
    equityCurve: combined,
    pnl,
    drawdown: maxDD,
  };
}

// ============================================================================
// earlyStop.ts — stop optimization if no improvement
// ============================================================================

export function earlyStop(
  scores: number[],
  patience: number
): boolean {
  if (scores.length <= patience) return false;

  const recent = scores.slice(-patience);
  const bestRecent = Math.max(...recent);
  const bestOverall = Math.max(...scores);

  return bestRecent < bestOverall;
}

export function toCSV(rows: any[]): string {
  if (rows.length === 0) return '';

  const headers = Object.keys(rows[0]);
  const lines = [headers.join(',')];

  for (const r of rows) {
    lines.push(headers.map(h => JSON.stringify(r[h])).join(','));
  }

  return lines.join('\n');
}

// ============================================================================
// eventBus.ts — real-time pub/sub
// ============================================================================

export type EventType =
  | 'tick'
  | 'analysis'
  | 'signal'
  | 'decision'
  | 'execution'
  | 'error';

type Callback = (data: any) => void;

export class EventBus {
  private listeners: Map<EventType, Set<Callback>> = new Map();

  on(type: EventType, cb: Callback) {
    if (!this.listeners.has(type)) {
      this.listeners.set(type, new Set());
    }
    this.listeners.get(type)!.add(cb);
  }

  emit(type: EventType, data: any) {
    const set = this.listeners.get(type);
    if (!set) return;
    for (const cb of set) cb(data);
  }
}

export const bus = new EventBus();

// liveDataFeed.ts — streaming ticks (deterministic mode optional)
// ============================================================================

import { bus } from './eventBus';

export interface Tick {
  symbol: string;
  price: number;
  volume: number;
  ts: number;
}

export class LiveDataFeed {
  private running = false;
  private price = 10000;
  private seed = 1;

  start() {
    this.running = true;
    this.loop();
  }

  stop() {
    this.running = false;
  }

  private nextRand() {
    this.seed = (1664525 * this.seed + 1013904223) >>> 0;
    return this.seed / 0xffffffff;
  }

  private loop() {
    if (!this.running) return;

    const r = this.nextRand() - 0.5;
    this.price = Math.max(1, Math.floor(this.price * (1 + r * 0.01)));

    const tick: Tick = {
      symbol: 'SYM',
      price: this.price,
      volume: Math.floor(this.nextRand() * 1000),
      ts: Date.now(),
    };

    bus.emit('tick', tick);

    setTimeout(() => this.loop(), 500);
  }
}

// snapshotStore.ts — periodic state snapshots
// ============================================================================

import { hashObject } from './deterministicHash';

export interface Snapshot {
  ts: string;
  state: any;
  hash: string;
}

export class SnapshotStore {
  private snaps: Snapshot[] = [];

  async take(state: any) {
    const hash = await hashObject(state);
    const snap: Snapshot = {
      ts: new Date().toISOString(),
      state,
      hash,
    };
    this.snaps.push(snap);
    return snap;
  }

  latest(): Snapshot | null {
    return this.snaps.length ? this.snaps[this.snaps.length - 1] : null;
  }

  all(): Snapshot[] {
    return this.snaps.slice();
  }
}

export const snapshotStore = new SnapshotStore();

// ============================================================================
// positionManager.ts — live positions + PnL
// ============================================================================

export interface LivePosition {
  symbol: string;
  shares: number;
  avgPrice: number;
}

export class PositionManager {
  private positions = new Map<string, LivePosition>();
  private cash = 100000;

  applyExecution(exec: any, price: number) {
    if (!exec) return;

    const { side, shares } = exec;

    if (side === 'buy') {
      const cost = shares * price;
      if (this.cash < cost) return;

      this.cash -= cost;

      const pos = this.positions.get('SYM') || {
        symbol: 'SYM',
        shares: 0,
        avgPrice: price,
      };

      const total = pos.shares + shares;
      pos.avgPrice =
        (pos.avgPrice * pos.shares + price * shares) / total;
      pos.shares = total;

      this.positions.set('SYM', pos);
    }

    if (side === 'sell') {
      const pos = this.positions.get('SYM');
      if (!pos || pos.shares < shares) return;

      this.cash += shares * price;
      pos.shares -= shares;

      if (pos.shares === 0) this.positions.delete('SYM');
    }
  }

  equity(price: number): number {
    let eq = this.cash;
    for (const p of this.positions.values()) {
      eq += p.shares * price;
    }
    return eq;
  }
}

export const positionManager = new PositionManager();

export function enforceRisk(params: {

  peak: number;
  drawdownLimit: number;

}

// ============================================================================
// consensusCluster.ts — multi-node agreement
// ============================================================================

export interface NodeDecision {
  nodeId: string;
  decision: any;
}

export function consensusCluster(
  decisions: NodeDecision[]
) {
  const votes: Record<string, number> = {};

  for (const d of decisions) {
    const key = JSON.stringify(d.decision);
    votes[key] = (votes[key] || 0) + 1;
  }

  let best = '';
  let max = 0;

  for (const k in votes) {
    if (votes[k] > max) {
      best = k;
      max = votes[k];
    }
  }

  return JSON.parse(best);
}

// commandRouter.ts — external command handling
// ============================================================================

import { validateKey } from './authLayer';
import { scalingManager } from './scalingManager';
import { failoverManager } from './failoverManager';

export function handleCommand(cmd: any) {
  if (!validateKey(cmd.key)) {
    return { error: 'unauthorized' };
  }

  switch (cmd.type) {
    case 'scale_up':
      scalingManager.scaleUp();
      return { workers: scalingManager.getWorkers() };

    case 'scale_down':
      scalingManager.scaleDown();
      return { workers: scalingManager.getWorkers() };

    case 'failover':
      failoverManager.failover();
      return { status: 'secondary_active' };

    case 'restore':
      failoverManager.restore();
      return { status: 'primary_restored' };

    default:
      return { error: 'unknown_command' };
  }
}

export interface Policy {
  buyBias: number;
  sellBias: number;
  holdBias: number;
}

export function defaultPolicy(): Policy {
  return {
    buyBias: 1,
    sellBias: 1,
    holdBias: 1,
  };
}

export function updatePolicy(
  policy: Policy,
  reward: number,
  action: string
): Policy {
  const lr = 0.05;

  if (action === 'buy') {
    policy.buyBias += lr * reward;
  } else if (action === 'sell') {
    policy.sellBias += lr * reward;
  } else {
    policy.holdBias += lr * reward;
  }

  return normalizePolicy(policy);
}

function normalizePolicy(p: Policy): Policy {
  const sum = p.buyBias + p.sellBias + p.holdBias || 1;

  return {
    buyBias: p.buyBias / sum,
    sellBias: p.sellBias / sum,
    holdBias: p.holdBias / sum,
  };
}

export function detectAnomalySeries(series: number[]): boolean {
  if (series.length < 5) return false;

  const mean =
    series.reduce((a, b) => a + b, 0) / series.length;

  const variance =
    series.reduce((s, x) => s + (x - mean) ** 2, 0) /
    series.length;

  return Math.sqrt(variance) > mean * 0.2;
}

// ============================================================================
// memoryGraph.ts — state-action graph (Markov structure)
// ============================================================================

export interface MemoryNode {
  stateHash: string;
  actions: Record<string, {
    count: number;
    totalReward: number;
  }>;
}

export class MemoryGraph {
  private graph: Map<string, MemoryNode> = new Map();

  update(stateHash: string, action: string, reward: number) {
    if (!this.graph.has(stateHash)) {
      this.graph.set(stateHash, {
        stateHash,
        actions: {},
      });
    }

    const node = this.graph.get(stateHash)!;

    if (!node.actions[action]) {
      node.actions[action] = { count: 0, totalReward: 0 };
    }

    node.actions[action].count++;
    node.actions[action].totalReward += reward;
  }

  bestAction(stateHash: string): string | null {
    const node = this.graph.get(stateHash);
    if (!node) return null;

    let best = null;
    let bestScore = -Infinity;

    for (const a in node.actions) {
      const data = node.actions[a];
      const avg = data.totalReward / data.count;

      if (avg > bestScore) {
        bestScore = avg;
        best = a;
      }
    }

    return best;
  }
}

export const memoryGraph = new MemoryGraph();

import { Policy } from './policyModel';

export interface PolicyAgent {
  id: string;
  policy: Policy;
  score: number;
}

export class PolicyEnsemble {
  private agents: PolicyAgent[] = [];

  add(agent: PolicyAgent) {
    this.agents.push(agent);
  }

  best(): PolicyAgent {
    return this.agents.sort((a, b) => b.score - a.score)[0];
  }

  updateScore(id: string, reward: number) {
    const agent = this.agents.find(a => a.id === id);
    if (agent) {
      agent.score += reward;
    }
  }

  all() {
    return this.agents;
  }
}

// ============================================================================
// strategyGenome.ts — evolving parameter sets
// ============================================================================

export interface Genome {
  id: string;
  params: Record<string, number>;
  fitness: number;
}

export function mutate(genome: Genome): Genome {
  const mutated = { ...genome.params };

  for (const k in mutated) {
    mutated[k] += (Math.random() - 0.5) * 0.1;
  }

  return {
    id: genome.id + '_mut',
    params: mutated,
    fitness: 0,
  };
}

export function crossover(a: Genome, b: Genome): Genome {
  const child: any = {};

  for (const k in a.params) {
    child[k] = Math.random() > 0.5 ? a.params[k] : b.params[k];
  }

  return {
    id: a.id + '_x_' + b.id,
    params: child,
    fitness: 0,
  };
}


// ============================================================================
// evolutionEngine.ts — Darwinian selection
// ============================================================================

import { Genome, mutate, crossover } from './strategyGenome';

export class EvolutionEngine {
  private population: Genome[] = [];

  seed(genomes: Genome[]) {
    this.population = genomes;
  }

  evolve() {
    this.population.sort((a, b) => b.fitness - a.fitness);

    const top = this.population.slice(0, 3);

    const next: Genome[] = [...top];

    for (let i = 0; i < top.length - 1; i++) {
      next.push(crossover(top[i], top[i + 1]));
      next.push(mutate(top[i]));
    }

    this.population = next;
  }

  best(): Genome {
    return this.population.sort((a, b) => b.fitness - a.fitness)[0];
  }
}

export const evolutionEngine = new EvolutionEngine();


// ============================================================================
// selfDiagnostics.ts — system introspection
// ============================================================================

export interface Diagnostics {
  performance: number;
  stability: number;
  anomalyRate: number;
}

export function runDiagnostics(params: {
  rewards: number[];
  anomalies: number;
}): Diagnostics {
  const avgReward =
    params.rewards.reduce((a, b) => a + b, 0) /
    (params.rewards.length || 1);

  const variance =
    params.rewards.reduce((s, r) => s + (r - avgReward) ** 2, 0) /
    (params.rewards.length || 1);

  return {
    performance: avgReward,
    stability: 1 / (1 + variance),
    anomalyRate: params.anomalies / (params.rewards.length || 1),
  };
}


// ============================================================================
// metaController.ts — system-level decision override
// ============================================================================

export function metaControl(params: {
  diagnostics: Diagnostics;
  policy: any;
}): any {
  const { diagnostics, policy } = params;

  if (diagnostics.anomalyRate > 0.3) {
    return {
      override: true,
      action: 'hold',
    };
  }

  if (diagnostics.performance < 0) {
    return {
      override: true,
      action: 'reduce_risk',
    };
  }

  return {
    override: false,
    policy,
  };
}


// ============================================================================
// curiosityEngine.ts — exploration vs exploitation
// ============================================================================

export function explorationDecision(
  bestAction: string | null,
  epsilon: number
): string {
  if (Math.random() < epsilon || !bestAction) {
    const actions = ['buy', 'sell', 'hold'];
    return actions[Math.floor(Math.random() * actions.length)];
  }

  return bestAction;
}

import { memoryGraph } from './memoryGraph';
import { policyEnsemble } from './policyEnsemble';
import { evolutionEngine } from './evolutionEngine';
import { runDiagnostics } from './selfDiagnostics';
import { metaControl } from './metaController';
import { explorationDecision } from './curiosityEngine';

export function runPhase6Extended(params: {
  stateHash: string;
  reward: number;
  action: string;
  historyRewards: number[];
  anomalies: number;
}) {
  // 1. Update memory graph
  memoryGraph.update(params.stateHash, params.action, params.reward);

  const bestAction = memoryGraph.bestAction(params.stateHash);

  // 2. Exploration vs exploitation
  const nextAction = explorationDecision(bestAction, 0.1);

  // 3. Diagnostics
  const diagnostics = runDiagnostics({
    rewards: params.historyRewards,
    anomalies: params.anomalies,
  });

  // 4. Meta control
  const control = metaControl({
    diagnostics,
    policy: nextAction,
  });

  // 5. Evolution step
  evolutionEngine.evolve();

  // 6. Policy ensemble update
  const agents = policyEnsemble.all();
  for (const a of agents) {
    policyEnsemble.updateScore(a.id, params.reward);
  }

  return {
    nextAction: control.override ? control.action : nextAction,
    diagnostics,
    control,
    bestGenome: evolutionEngine.best(),
  };
}
